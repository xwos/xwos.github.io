42{
43 xwsz_t num_max, n, nm;
44 xwptr_t curr, next;
45 xwer_t rc;
46
47 XWOS_VALIDATE((msa), "nullptr", -EFAULT);
48
49 card_size = XWBOP_ALIGN(card_size, XWMM_ALIGNMENT);
50 num_max = total_size / card_size;
51 if (0 == num_max) {
52 rc = -E2SMALL;
53 goto err_mem2small;
54 }
55
56 msa->zone.origin = origin;
57 msa->zone.size = total_size;
58 msa->name = name;
59 msa->card_size = card_size;
60 msa->num_max = num_max;
61 msa->num_free = num_max;
62 msa->ctor = ctor;
63 msa->dtor = dtor;
64
65 /* 构造所有对象 */
66 if (ctor) {
67 curr = origin;
68 next = curr;
69 for (n = 0; n < num_max; n++) {
70 next += card_size;
71 ctor((void *)curr);
72 curr = next;
73 }
74 msa->backup = *((xwptr_t *)origin);
75 } else {
76 msa->backup = 0;
77 }
78
79 xwlib_lfq_init(&msa->free_list);
80 curr = origin;
81 next = origin;
82 nm = num_max;
83 for (n = 0; n < nm; n++) {
84 next += card_size;
85 xwlib_lfq_init((atomic_xwlfq_t *)curr);
86 xwlib_lfq_push(&msa->free_list, (atomic_xwlfq_t *)curr);
87 curr = next;
88 }
89
90 return XWOK;
91
92err_mem2small:
93 return rc;
94}

108__xwos_api
109xwer_t xwmm_memslice_alloc(struct xwmm_memslice * msa, void ** membuf)
110{
111 xwlfq_t * card;
112 xwer_t rc;
113
114 XWOS_VALIDATE((msa), "nullptr", -EFAULT);
115 XWOS_VALIDATE((membuf), "nullptr", -EFAULT);
116
117 card = xwlib_lfq_pop(&msa->free_list);
118 if (NULL == card) {
119 rc = -ENOMEM;
120 *membuf = NULL;
121 goto err_lfq_pop;
122 }
123 xwaop_sub(xwsz_t, &msa->num_free, 1, NULL, NULL);
124 *(xwptr_t *)card = msa->backup; /* restore original data */
125 *membuf = card;
126 return XWOK;
127
128err_lfq_pop:
129 return rc;
130}

142__xwos_api
143xwer_t xwmm_memslice_free(struct xwmm_memslice * msa, void * mem)
144{
145 xwer_t rc;
146 atomic_xwlfq_t * card;
147
148 XWOS_VALIDATE((msa), "nullptr", -EFAULT);
149 XWOS_VALIDATE((mem), "nullptr", -EFAULT);
150
151 if (__xwcc_unlikely((xwptr_t)mem < msa->zone.origin ||
152 ((xwptr_t)mem - msa->zone.origin) > msa->zone.size)) {
153 rc = -EOWNER;
154 } else {
155 if (msa->dtor) {
156 msa->dtor(mem);
157 }/* else {} */
158 card = (atomic_xwlfq_t *)mem;
159 xwlib_lfq_push(&msa->free_list, card);
160 xwaop_add(xwsz_t, &msa->num_free, 1, NULL, NULL);
161 rc = XWOK;
162 }
163
164 return rc;
165}
