157__xwmd_api
158xwer_t xwcq_eq(struct xwcq * cq, const xwu8_t * data, xwsz_t * size)
159{
160 xwreg_t cpuirq;
161 xwsz_t eqsize;
162
163 XWOS_VALIDATE((cq), "nullptr", -EFAULT);
164 XWOS_VALIDATE((data), "nullptr", -EFAULT);
165 XWOS_VALIDATE((size), "nullptr", -EFAULT);
166
167 eqsize = *size;
168 eqsize = eqsize > cq->slotsize ? cq->slotsize : eqsize;
169 xwos_splk_lock_cpuirqsv(&cq->lock, &cpuirq);
170 memcpy(&cq->q[(xwsz_t)cq->rear * cq->slotsize], data, eqsize);
171 cq->rear++;
172 if (cq->rear >= (xwssq_t)cq->slotnum) {
173 cq->rear = 0;
174 }
175 if (cq->rear == cq->front) {
176 /* Queue is overflow. Discard the oldest data */
177 cq->front++;
178 if (cq->front >= (xwssq_t)cq->slotnum) {
179 cq->front = 0;
180 }
181 }
182 xwos_splk_unlock_cpuirqrs(&cq->lock, cpuirq);
183 xwos_sem_post(&cq->sem);
184 *size = eqsize;
185 return XWOK;
186}
