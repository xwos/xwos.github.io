88static __xwmd_code
89xwer_t xwmq_activate(struct xwmq * mq,
90 struct xwmq_msg * txq, xwsz_t num,
91 xwobj_gc_f gcfunc)
92{
93 xwer_t rc;
94
95 rc = xwos_object_activate(&mq->xwobj, gcfunc);
96 if (__xwcc_unlikely(rc < 0)) {
97 goto err_xwobj_activate;
98 }
99 xwmm_memslice_init(&mq->txq,
100 (xwptr_t)txq, sizeof(struct xwmq_msg) * num,
101 sizeof(struct xwmq_msg), xwmq_txq_name,
102 xwmq_msg_construct, NULL);
103 xwos_sem_init(&mq->txqsem, (xwssq_t)num, (xwssq_t)num);
104 xwlib_bclst_init_head(&mq->rxq);
105 xwos_splk_init(&mq->rxqlock);
106 xwos_sem_init(&mq->rxqsem, 0, (xwssq_t)num);
107 return XWOK;
108
109err_xwobj_activate:
110 return rc;
111}

174__xwmd_api
175xwer_t xwmq_eq_to(struct xwmq * mq, xwsq_t topic, void * data, xwtm_t to)
176{
177 xwer_t rc;
178 xwreg_t cpuirq;
179 struct xwmq_msg * msg;
180
181 XWOS_VALIDATE((mq), "nullptr", -EFAULT);
182 XWOS_VALIDATE((data), "nullptr", -EFAULT);
183
184 rc = xwos_sem_wait_to(&mq->txqsem, to);
185 if (rc < 0) {
186 goto err_sem_wait_to;
187 }
188 msg = xwmq_msg_get(mq);
189 msg->topic = topic;
190 msg->data = data;
191 xwos_splk_lock_cpuirqsv(&mq->rxqlock, &cpuirq);
192 xwlib_bclst_add_tail(&mq->rxq, &msg->node);
193 xwos_splk_unlock_cpuirqrs(&mq->rxqlock, cpuirq);
194 xwos_sem_post(&mq->rxqsem);
195 return XWOK;
196
197err_sem_wait_to:
198 return rc;
199}
