59xwer_t xwpcp_start(struct xwpcp * xwpcp, const char * name,
60 const struct xwpcp_hwifal_operation * hwifops, void * hwifcb,
61 xwu8_t * mem, xwsz_t memsize)
62{
63 struct xwos_thd_attr attr;
64 xwer_t rc;
65 xwssq_t i, j;
66
67 XWPCP_VALIDATE((xwpcp), "nullptr", -EFAULT);
68 XWPCP_VALIDATE((hwifops), "nullptr", -EFAULT);
69 XWPCP_VALIDATE((hwifops->tx), "nullptr", -EFAULT);
70 XWPCP_VALIDATE((hwifops->rx), "nullptr", -EFAULT);
71
72 XWPCP_VALIDATE_FORCE((mem), "nullptr", -EFAULT);
73 XWPCP_VALIDATE_FORCE((XWPCP_MEMPOOL_SIZE == memsize),
74 "memsize-error",
75 -ESIZE);
76
77 xwpcplogf(DEBUG, "Starting XWPCP-%s ...\n", XWPCP_VERSION);
78
79 xwos_object_construct(&xwpcp->xwobj);
80 xwpcp->hwifst = XWPCP_HWIFST_CLOSED;
81 xwpcp->hwifcb = NULL;
82 xwpcp->name = name;
83 xwpcp->hwifops = hwifops;
84 xwpcp->txthd = XWOS_THD_NILD;
85 xwpcp->rxthd = XWOS_THD_NILD;
86 xwpcp->mempool = NULL;
87 xwos_object_activate(&xwpcp->xwobj, xwpcp_gc);
88
89 /* 创建内存池 */
90 xwpcp->mempool = (struct xwmm_bma *)&xwpcp->mempool_bma_raw;
91 rc = xwmm_bma_init(xwpcp->mempool, "xwpcp.mempool",
92 (xwptr_t)mem, memsize,
93 XWPCP_MEMBLK_SIZE, XWPCP_MEMBLK_ODR);
94 if (__xwcc_unlikely(rc < 0)) {
95 xwpcplogf(ERR, "Init bma ... [rc:%d]\n", rc);
96 goto err_bma_init;
97 }
98 xwpcplogf(DEBUG, "Create BMA ... [OK]\n");
99
100 /* 初始化发送状态机 */
101 xwpcp->txq.cnt = 0;
102 for (i = 0; i < (xwssq_t)XWPCP_MEMBLK_NUM; i++) {
103 xwlib_bclst_init_node(&xwpcp->txq.car[i].node);
104 xwpcp->txq.car[i].state = XWPCP_CRS_IDLE;
105 xwpcp->txq.car[i].idx = i;
106 xwpcp->txq.car[i].pri = XWPCP_INVALID_PRI;
107 xwpcp->txq.car[i].slot = NULL;
108 }
109 memset((void *)&xwpcp->txq.carbmp, 0, sizeof(xwpcp->txq.carbmp));
110 for (i = 0; i < (xwssq_t)XWPCP_PRI_NUM; i++) {
111 xwlib_bclst_init_head(&xwpcp->txq.q[i]);
112 }
113 memset(&xwpcp->txq.qnebmp, 0, sizeof(xwpcp->txq.qnebmp));
114 xwos_splk_init(&xwpcp->txq.qlock);
115 rc = xwos_sem_init(&xwpcp->txq.qsem, 0, XWPCP_MEMBLK_NUM);
116 if (__xwcc_unlikely(rc < 0)) {
117 xwpcplogf(ERR, "Init TXQ semaphore ... [rc:%d]\n", rc);
118 goto err_txqsem_init;
119 }
120 rc = xwos_mtx_init(&xwpcp->txq.csmtx, XWOS_SKD_PRIORITY_RT_MIN);
121 if (__xwcc_unlikely(rc < 0)) {
122 xwpcplogf(ERR, "Init xwpcp->csmtx ... [rc:%d]\n", rc);
123 goto err_csmtx_init;
124 }
125 rc = xwos_cond_init(&xwpcp->txq.cscond);
126 if (__xwcc_unlikely(rc < 0)) {
127 xwpcplogf(ERR, "Init xwpcp->cscond ... [rc:%d]\n", rc);
128 goto err_cscond_init;
129 }
130 xwpcp->txq.remote.ack = 0;
131 xwpcp->txq.remote.id = 0;
132 xwpcp->txq.tmp = NULL;
133 xwos_splk_init(&xwpcp->txq.notiflk);
134
135 /* 初始化接收状态机 */
136 xwpcp->rxq.cnt = 0;
137 for (i = 0; i < (xwssq_t)XWPCP_PORT_NUM; i++) {
138 xwlib_bclst_init_head(&xwpcp->rxq.q[i]);
139 xwos_splk_init(&xwpcp->rxq.lock[i]);
140 rc = xwos_sem_init(&xwpcp->rxq.sem[i], 0, XWPCP_MEMBLK_NUM);
141 if (__xwcc_unlikely(rc < 0)) {
142 xwpcplogf(ERR, "Init RXQ semaphore[%d] ... [rc:%d]\n",
143 i, rc);
144 goto err_rxqsem_init;
145 }
146 }
147
148 /* 打开硬件接口 */
149 rc = xwpcp_hwifal_open(xwpcp, hwifcb);
150 if (rc < 0) {
151 goto err_hwifal_open;
152 }
153
154 /* 创建线程 */
155 xwos_thd_attr_init(&attr);
156 attr.name = "xwmd.isc.xwpcp.rxthd";
157 attr.stack = NULL;
158 attr.stack_size = 2048;
159 attr.priority = XWPCP_RXTHD_PRIORITY;
160 attr.detached = false;
161 attr.privileged = true;
162 rc = xwos_thd_create(&xwpcp->rxthd, &attr, (xwos_thd_f)xwpcp_rxthd, xwpcp);
163 if (rc < 0) {
164 goto err_rxthd_create;
165 }
166
167 xwos_thd_attr_init(&attr);
168 attr.name = "xwmd.isc.xwpcp.txthd";
169 attr.stack = NULL;
170 attr.stack_size = 2048;
171 attr.priority = XWPCP_TXTHD_PRIORITY;
172 attr.detached = false;
173 attr.privileged = true;
174 rc = xwos_thd_create(&xwpcp->txthd, &attr, (xwos_thd_f)xwpcp_txthd, xwpcp);
175 if (rc < 0) {
176 goto err_txthd_create;
177 }
178
179 return XWOK;
180
181err_txthd_create:
182 xwos_thd_detach(xwpcp->rxthd);
183 xwos_thd_quit(xwpcp->rxthd);
184 xwpcp->rxthd = XWOS_THD_NILD;
185err_rxthd_create:
186 xwpcp_hwifal_close(xwpcp);
187err_hwifal_open:
188err_rxqsem_init:
189 for (j = i - 1; j >= 0; j--) {
190 xwos_sem_fini(&xwpcp->rxq.sem[j]);
191 }
192 xwos_cond_fini(&xwpcp->txq.cscond);
193err_cscond_init:
194 xwos_mtx_fini(&xwpcp->txq.csmtx);
195err_csmtx_init:
196 xwos_sem_fini(&xwpcp->txq.qsem);
197err_txqsem_init:
198err_bma_init:
199 xwos_object_rawput(&xwpcp->xwobj);
200 return rc;
201}

227static __xwmd_code
228xwer_t xwpcp_gc(void * obj)
229{
230 struct xwpcp * xwpcp;
231 struct xwpcp_carrier * car;
232 union xwpcp_slot * slot;
233 xwer_t rc, childrc;
234 xwssq_t j;
235
236 xwpcp = obj;
237 if (xwpcp->txthd.thd) {
238 rc = xwos_thd_stop(xwpcp->txthd, &childrc);
239 if (XWOK == rc) {
240 xwpcp->txthd = XWOS_THD_NILD;
241 xwpcplogf(INFO, "Stop XWPCP TX thread... [OK]\n");
242 }
243 }
244
245 if (xwpcp->rxthd.thd) {
246 rc = xwos_thd_stop(xwpcp->rxthd, &childrc);
247 if (XWOK == rc) {
248 xwpcp->rxthd = XWOS_THD_NILD;
249 xwpcplogf(INFO, "Stop XWPCP RX thread... [OK]\n");
250 }
251 }
252
253 xwpcp_hwifal_close(xwpcp);
254
255 /* 释放TXQ中的剩余帧 */
256 do {
257 rc = xwos_sem_trywait(&xwpcp->txq.qsem);
258 if (rc < 0) {
259 break;
260 }
261 car = xwpcp_txq_choose(xwpcp);
262 xwmm_bma_free(xwpcp->mempool, car->slot);
263 car->slot = NULL;
264 xwpcp_txq_carrier_free(xwpcp, car);
265 } while (true);
266
267 /* 释放RXQ中的剩余帧 */
268 for (j = XWPCP_PORT_NUM - 1; j >= 0; j--) {
269 do {
270 rc = xwos_sem_trywait(&xwpcp->rxq.sem[j]);
271 if (rc < 0) {
272 break;
273 }
274 slot = xwpcp_rxq_choose(xwpcp, (xwu8_t)j);
275 xwmm_bma_free(xwpcp->mempool, slot);
276 } while (true);
277 }
278
279 for (j = XWPCP_PORT_NUM - 1; j >= 0; j--) {
280 xwos_sem_fini(&xwpcp->rxq.sem[j]);
281 }
282 xwos_cond_fini(&xwpcp->txq.cscond);
283 xwos_mtx_fini(&xwpcp->txq.csmtx);
284 xwos_sem_fini(&xwpcp->txq.qsem);
285 return XWOK;
286}

350__xwmd_api
351xwer_t xwpcp_tx(struct xwpcp * xwpcp,
352 const xwu8_t data[], xwsz_t * size,
353 xwu8_t pri, xwu8_t port, xwu8_t qos,
354 xwtm_t to)
355{
356 struct xwpcp_txcb_arg cbarg;
357 union xwos_ulock ulk;
358 xwpcp_txh_t txh;
359 xwsq_t lkst;
360 xwer_t rc;
361
362 XWPCP_VALIDATE((xwpcp), "nullptr", -EFAULT);
363 XWPCP_VALIDATE((data), "nullptr", -EFAULT);
364 XWPCP_VALIDATE((size), "nullptr", -EFAULT);
365 XWPCP_VALIDATE((*size <= XWPCP_SDU_MAX_SIZE), "size-invalid", -E2BIG);
366 XWPCP_VALIDATE((pri < XWPCP_PRI_NUM), "pri-invalid", -E2BIG);
367 XWPCP_VALIDATE((port > XWPCP_PORT_CMD), "port0-not-permitted", -ENXIO);
368 XWPCP_VALIDATE((port < XWPCP_PORT_NUM), "no-such-port", -ENODEV);
369 XWPCP_VALIDATE((qos < XWPCP_MSG_QOS_NUM), "qos-invalid", -EINVAL);
370
371 rc = xwpcp_grab(xwpcp);
372 if (__xwcc_unlikely(rc < 0)) {
373 rc = -EPERM;
374 goto err_ifnotrdy;
375 }
376
377 xwos_splk_init(&cbarg.splk);
378 xwos_cond_init(&cbarg.cond);
379 cbarg.rc = -EINPROGRESS;
380
381 rc = xwpcp_eq_msg(xwpcp, data, *size, pri, port, qos,
382 xwpcp_txcb_notify, &cbarg, &txh);
383 if (__xwcc_unlikely(rc < 0)) {
384 goto err_xwpcp_eq;
385 }
386
387 ulk.osal.splk = &cbarg.splk;
388 xwos_splk_lock(&cbarg.splk);
389 if (-EINPROGRESS == cbarg.rc) {
390 rc = xwos_cond_wait_to(&cbarg.cond, ulk, XWOS_LK_SPLK, NULL, to, &lkst);
391 if (XWOK == rc) {
392 rc = cbarg.rc;
393 xwos_splk_unlock(&cbarg.splk);
394 } else {
395 if (XWOS_LKST_LOCKED == lkst) {
396 xwos_splk_unlock(&cbarg.splk);
397 }/* else {} */
398 xwos_splk_lock(&xwpcp->txq.notiflk);
399 xwos_splk_lock(&cbarg.splk);
400 if (-EINPROGRESS == cbarg.rc) {
401 txh->slot->tx.ntfcb = NULL;
402 txh->slot->tx.cbarg = NULL;
403 } else {
404 rc = cbarg.rc;
405 }
406 xwos_splk_unlock(&cbarg.splk);
407 xwos_splk_unlock(&xwpcp->txq.notiflk);
408 }
409 } else {
410 rc = cbarg.rc;
411 xwos_splk_unlock(&cbarg.splk);
412 }
413 xwos_cond_fini(&cbarg.cond);
414 xwpcp_put(xwpcp);
415 return rc;
416
417err_xwpcp_eq:
418 xwos_cond_fini(&cbarg.cond);
419 xwpcp_put(xwpcp);
420err_ifnotrdy:
421 return rc;
422}

556xwer_t xwpcp_rx(struct xwpcp * xwpcp, xwu8_t port,
557 xwu8_t rxbuf[], xwsz_t * size, xwu8_t * qos,
558 xwtm_t to)
559{
560 union xwpcp_slot * slot;
561 xwu8_t * sdupos;
562 xwsz_t bufsize;
563 xwsz_t sdusize;
564 xwsz_t realsize;
565 xwer_t rc;
566
567 XWPCP_VALIDATE((xwpcp), "nullptr", -EFAULT);
568 XWPCP_VALIDATE((rxbuf), "nullptr", -EFAULT);
569 XWPCP_VALIDATE((size), "nullptr", -EFAULT);
570 XWPCP_VALIDATE((port > XWPCP_PORT_CMD), "port0-not-permitted", -ENXIO);
571 XWPCP_VALIDATE((port < XWPCP_PORT_NUM), "no-such-port", -ENODEV);
572
573 rc = xwpcp_grab(xwpcp);
574 if (__xwcc_unlikely(rc < 0)) {
575 rc = -EPERM;
576 goto err_ifnotrdy;
577 }
578
579 bufsize = *size;
580 xwpcplogf(DEBUG, "port:%d, buffer size:0x%X\n", port, bufsize);
581 rc = xwos_sem_wait_to(&xwpcp->rxq.sem[port], to);
582 if (__xwcc_unlikely(rc < 0)) {
583 goto err_sem_wait_to;
584 }
585
586 slot = xwpcp_rxq_choose(xwpcp, port);
587 sdupos = XWPCP_SDUPOS(&slot->rx.frm.head);
588 sdusize = XWPCP_RXSDUSIZE(&slot->rx);
589
590 if (bufsize < sdusize) {
591 realsize = bufsize;
592 xwpcplogf(WARNING,
593 "buffer size is too small(0x%X)!, expected size is 0x%X\n",
594 bufsize, sdusize);
595 } else {
596 realsize = sdusize;
597 }
598 memcpy(rxbuf, sdupos, realsize);
599 if (qos) {
600 *qos = slot->rx.frm.head.qos;
601 }
602 *size = realsize;
603
604 xwmm_bma_free(xwpcp->mempool, slot);
605 xwpcp_put(xwpcp);
606 return XWOK;
607
608err_sem_wait_to:
609 xwpcp_put(xwpcp);
610err_ifnotrdy:
611 return rc;
612}
