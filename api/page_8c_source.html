78__xwos_api
79xwer_t xwmm_mempool_page_allocator_init(struct xwmm_mempool_page_allocator * pa,
80 const char * name,
81 xwptr_t origin, xwsz_t size, xwsz_t pgsize,
82 struct xwmm_mempool_page_odrbtree * odrbtree,
83 struct xwmm_mempool_page * pgarray)
84{
85 xwsz_t i;
86 xwssq_t order;
87 xwsz_t nr;
88 xwer_t rc;
89
90 if (__xwcc_unlikely(size < pgsize)) {
91 rc = -E2SMALL;
92 goto err_mem2small;
93 }
94 if (pgsize & XWMM_UNALIGNED_MASK) {
95 rc = -EALIGN;
96 goto err_aligned;
97 }
98 nr = size / pgsize;
99 order = xwbop_fls(xwsz_t, nr);
100 if ((order < 0) || (nr & ((1U << order) - 1U))) {
101 rc = -EALIGN;
102 goto err_aligned;
103 }
104
105 pa->zone.origin = origin;
106 pa->zone.size = size;
107 pa->name = name;
108 pa->pgsize = pgsize;
109 pa->max_order = (xwsq_t)order;
110 pa->odrbtree = odrbtree;
111 pa->pgarray = pgarray;
112
113 for (i = 0; i <= pa->max_order; i++) {
114 xwlib_rbtree_init(&pa->odrbtree[i].tree);
115 pa->odrbtree[i].leftmost = NULL;
116 pa->odrbtree[i].owner = pa;
117 pa->odrbtree[i].order = i;
118 xwos_sqlk_init(&pa->odrbtree[i].lock);
119 }
120 for (i = 0; i < nr; i++) {
121 pa->pgarray[i].order = XWMM_MEMPOOL_PAGE_ORDER_CMB;
122 pa->pgarray[i].mapping = XWMM_MEMPOOL_PAGE_MAPPING_INVAILD;
123 pa->pgarray[i].attr.free.seq = i;
124 xwlib_rbtree_init_node(&pa->pgarray[i].attr.free.rbnode);
125 pa->pgarray[i].data.value = 0;
126 }
127 xwmm_mempool_page_odrbtree_add(&pa->odrbtree[pa->max_order],
128 &pa->pgarray[0]);
129
130 pa->i_a.malloc = xwmm_mempool_page_i_a_malloc;
131 pa->i_a.free = xwmm_mempool_page_i_a_free;
132
133 return XWOK;
134
135err_aligned:
136err_mem2small:
137 return rc;
138}

149__xwos_api
150xwer_t xwmm_mempool_page_allocate(struct xwmm_mempool_page_allocator * pa,
151 xwsq_t order,
152 struct xwmm_mempool_page ** pgbuf)
153{
154 xwsq_t odr;
155 struct xwmm_mempool_page_odrbtree * ot;
156 struct xwmm_mempool_page * pg;
157 xwptr_t origin;
158 xwer_t rc;
159
160 rc = -ENOMEM;
161 for (odr = order; odr <= pa->max_order; odr++) {
162 ot = &pa->odrbtree[odr];
163 pg = xwmm_mempool_page_odrbtree_choose(ot);
164 if (!pg) {
165 continue;
166 }
167 xwmm_mempool_page_divide_page(pa, pg, order, ot);
168 origin = pa->zone.origin + ((xwsq_t)pg->attr.free.seq * pa->pgsize);
169 pg->mapping = origin;
170 *pgbuf = pg;
171 rc = XWOK;
172 break;
173 }
174 return rc;
175}

187 struct xwmm_mempool_page * pg)

190

192 rc = -EALREADY;

196 pg->attr.free.seq = xwmm_mempool_page_get_seq(pa, pg);

215static __xwos_code
216void xwmm_mempool_page_divide_page(struct xwmm_mempool_page_allocator * pa,
217 struct xwmm_mempool_page * pg,
218 xwsq_t target_odr,
219 struct xwmm_mempool_page_odrbtree * ot)
220{
221 xwsq_t pg_odr;
222 xwsq_t pg_seq;
223 xwsq_t pg_offset;
224 struct xwmm_mempool_page * buddy;
225
226 pg_odr = pg->order;
227 pg_seq = pg->attr.free.seq;
228 pg_offset = 1U << pg_odr;
229 while (pg_odr > target_odr) {
230 ot--;
231 pg_odr--;
232 pg_offset >>= 1;
233 buddy = &pa->pgarray[pg_seq + pg_offset];
234 xwmm_mempool_page_odrbtree_add(ot, buddy);
235 pg->order = pg_odr;
236 }
237}

246static __xwos_code
247void xwmm_mempool_page_combine(struct xwmm_mempool_page_allocator * pa,
248 struct xwmm_mempool_page * pg)
249{
250 struct xwmm_mempool_page * buddy;
251 xwsq_t odr;
252 xwer_t rc;
253
254 odr = pg->order + 1U;
255 while (odr < pa->max_order) {
256 buddy = xwmm_mempool_page_get_buddy(pa, pg);
257 if (buddy->order != pg->order) {
258 break;
259 }
260 /* 不能使用buddy->order，因为未上锁时，buddy->order有可能改变，
261 尤其是多核系统，但此时pg还未加入到红黑树，是私有数据。 */
262 rc = xwmm_mempool_page_odrbtree_remove(&pa->odrbtree[pg->order],
263 buddy);
264 if (rc < 0) {
265 break;
266 }
267 if (buddy > pg) {
268 buddy->order = XWMM_MEMPOOL_PAGE_ORDER_CMB;
269 pg->order = odr;
270 } else {
271 pg->order = XWMM_MEMPOOL_PAGE_ORDER_CMB;
272 buddy->order = odr;
273 pg = buddy;
274 }
275 odr++;
276 }
277 odr = pg->order;
278 xwmm_mempool_page_odrbtree_add(&pa->odrbtree[odr], pg);
279}

292static __xwos_code
293xwer_t xwmm_mempool_page_odrbtree_add(struct xwmm_mempool_page_odrbtree * ot,
294 struct xwmm_mempool_page * pg)
295{
296 struct xwlib_rbtree * tree;
297 struct xwlib_rbtree_node ** new;
298 xwptr_t lpc;
299 struct xwlib_rbtree_node * rbn;
300 xwsq_t pgseq;
301 struct xwmm_mempool_page * b, * leftmost;
302 xwreg_t flag;
303 xwer_t rc;
304
305 tree = &ot->tree;
306 pgseq = pg->attr.free.seq;
307 if (!pg->mapping) {
308 rc = -EEXIST;
309 goto err_exist;
310 }
311
312 pg->order = ot->order;
313 xwos_sqlk_wr_lock_cpuirqsv(&ot->lock, &flag);
314 pg->mapping = XWMM_MEMPOOL_PAGE_MAPPING_FREE;
315 new = &tree->root;
316 lpc = (xwptr_t)new;
317 leftmost = ot->leftmost;
318 if (NULL == leftmost) { /* rbtree is empty. */
319 ot->leftmost = pg;
320 } else if (pgseq < leftmost->attr.free.seq) {
321 new = &(leftmost->attr.free.rbnode.left);
322 lpc = (xwptr_t)new;
323 ot->leftmost = pg;
324 } else {
325 rbn = *new;
326 while (rbn) {
327 b = xwlib_rbtree_entry(rbn, struct xwmm_mempool_page,
328 attr.free.rbnode);
329 if (pgseq < b->attr.free.seq) {
330 new = &rbn->left;
331 lpc = (xwptr_t)new;
332 rbn = rbn->left;
333 } else if (pgseq > b->attr.free.seq) {
334 new = &rbn->right;
335 lpc = (xwptr_t)new | XWLIB_RBTREE_POS_RIGHT;
336 rbn = rbn->right;
337 } else {
338 XWOS_BUG();
339 }
340 }
341 }
342 xwlib_rbtree_link(&pg->attr.free.rbnode, lpc);
343 xwlib_rbtree_insert_color(tree, &pg->attr.free.rbnode);
344 xwos_sqlk_wr_unlock_cpuirqrs(&ot->lock, flag);
345 return XWOK;
346
347err_exist:
348 return rc;
349}

361static __xwos_code
362xwer_t xwmm_mempool_page_odrbtree_remove(struct xwmm_mempool_page_odrbtree * ot,
363 struct xwmm_mempool_page * pg)
364{
365 struct xwlib_rbtree * tree;
366 struct xwlib_rbtree_node * s;
367 xwreg_t flag;
368 xwer_t rc;
369
370 tree = &ot->tree;
371 if (pg->order != ot->order) {
372 rc = -ESRCH;
373 goto err_odrerr;
374 }
375
376 xwos_sqlk_wr_lock_cpuirqsv(&ot->lock, &flag);
377 if (pg->mapping) {
378 rc = -EBUSY;
379 goto err_notfree;
380 }
381 pg->mapping = XWMM_MEMPOOL_PAGE_MAPPING_INVAILD;
382 if (pg == ot->leftmost) {
383 s = pg->attr.free.rbnode.right;
384 if (!s) {
385 s = xwlib_rbtree_get_parent(&pg->attr.free.rbnode);
386 }
387 if (s != (struct xwlib_rbtree_node *)&tree->root) {
388 ot->leftmost = xwlib_rbtree_entry(s, struct xwmm_mempool_page,
389 attr.free.rbnode);
390 } else {
391 ot->leftmost = NULL;
392 }
393 }
394 xwlib_rbtree_remove(tree, &pg->attr.free.rbnode);
395 xwos_sqlk_wr_unlock_cpuirqrs(&ot->lock, flag);
396 xwlib_rbtree_init_node(&pg->attr.free.rbnode);
397 return XWOK;
398
399err_notfree:
400 xwos_sqlk_wr_unlock_cpuirqrs(&ot->lock, flag);
401err_odrerr:
402 return rc;
403}

412static __xwos_code
413struct xwmm_mempool_page *
414xwmm_mempool_page_odrbtree_choose(struct xwmm_mempool_page_odrbtree * ot)
415{
416 struct xwmm_mempool_page * leftmost;
417 xwreg_t flag;
418 struct xwlib_rbtree * tree;
419 struct xwlib_rbtree_node * s;
420
421 tree = &ot->tree;
422 xwos_sqlk_wr_lock_cpuirqsv(&ot->lock, &flag);
423 leftmost = ot->leftmost;
424 if (!leftmost) {
425 xwos_sqlk_wr_unlock_cpuirqrs(&ot->lock, flag);
426 goto err_empty;
427 }
428 leftmost->mapping = XWMM_MEMPOOL_PAGE_MAPPING_INVAILD;
429 s = leftmost->attr.free.rbnode.right;
430 if (!s) {
431 s = xwlib_rbtree_get_parent(&leftmost->attr.free.rbnode);
432 }
433 if (s != (struct xwlib_rbtree_node *)&tree->root) {
434 ot->leftmost = xwlib_rbtree_entry(s, struct xwmm_mempool_page,
435 attr.free.rbnode);
436 } else {
437 ot->leftmost = NULL;
438 }
439 xwlib_rbtree_remove(tree, &leftmost->attr.free.rbnode);
440 xwos_sqlk_wr_unlock_cpuirqrs(&ot->lock, flag);
441
442 xwlib_rbtree_init_node(&leftmost->attr.free.rbnode);
443 return leftmost;
444
445err_empty:
446 return NULL;
447}

494xwmm_mempool_page_get_buddy(struct xwmm_mempool_page_allocator * pa,
495 struct xwmm_mempool_page * pg)
496{

538static __xwos_code
539xwer_t xwmm_mempool_page_i_a_malloc(void * this, xwsz_t size, void ** membuf)
540{
541 struct xwmm_mempool_page_allocator * pa;
542 struct xwmm_mempool_page * pg;
543 xwssq_t order;
544 xwsz_t nr;
545 xwer_t rc;
546
547 pa = this;
548 nr = size / pa->pgsize;
549 order = xwbop_fls(xwsz_t, nr);
550 if ((order < 0) || ((pa->pgsize << order) < size)) {
551 order++;
552 }
553 rc = xwmm_mempool_page_allocate(pa, (xwsq_t)order, &pg);
554 if (XWOK == rc) {
555 pg->data.value = size;
556 *membuf = (void *)pg->mapping;
557 }
558 return rc;
559}

569static __xwos_code
570xwer_t xwmm_mempool_page_i_a_free(void * this, void * mem)
571{
572 struct xwmm_mempool_page_allocator * pa;
573 struct xwmm_mempool_page * pg;
574 xwer_t rc;
575
576 pa = this;
577
578 if (((xwptr_t)mem < pa->zone.origin) ||
579 (((xwptr_t)mem - pa->zone.origin) >= (xwptr_t)pa->zone.size)) {
580 rc = -ERANGE;
581 goto err_range;
582 }
583
584 pg = xwmm_mempool_mem_to_page(pa, mem);
585 rc = xwmm_mempool_page_free(pa, pg);
586 if (rc < 0) {
587 goto err_page_free;
588 }
589 return XWOK;
590
591err_page_free:
592err_range:
593 return rc;
