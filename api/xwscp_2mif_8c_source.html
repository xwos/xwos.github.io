75 XWSCP_VALIDATE((xwscp), "nullptr", -EFAULT);

198xwer_t xwscp_gc(void * obj)
199{
200 struct xwscp * xwscp;
201 xwer_t rc, childrc;
202
203 xwscp = obj;
204 if (xwscp->rx.thd.thd) {
205 rc = xwos_thd_stop(xwscp->rx.thd, &childrc);
206 if (XWOK == rc) {
207 xwscp->rx.thd = XWOS_THD_NILD;

209 }
210 }
211

213 xwos_sem_fini(&xwscp->rx.sem);
214 xwos_cond_fini(&xwscp->tx.cscond);
215 xwos_mtx_fini(&xwscp->tx.csmtx);
216 xwos_mtx_fini(&xwscp->tx.mtx);
217 return XWOK;

220static __xwmd_code
221xwer_t xwscp_connect_once(struct xwscp * xwscp, xwtm_t to)
222{
223 union xwos_ulock ulk;
224 xwsq_t lkst;
225 xwtm_t next;
226 xwer_t rc;
227
228 ulk.osal.mtx = &xwscp->tx.csmtx;
229 rc = xwos_mtx_lock_to(&xwscp->tx.csmtx, to);
230 if (__xwcc_unlikely(rc < 0)) {
231 goto err_mtx_lock_to;
232 }
233 xwaop_c0m(xwsq_t, &xwscp->hwifst, XWSCP_HWIFST_CONNECT, NULL, NULL);
234 rc = xwscp_tx_cmd_connect(xwscp);
235 if (__xwcc_unlikely(rc < 0)) {
236 goto err_tx;
237 }
238 next = xwtm_ft(XWSCP_RETRY_PERIOD);
239 if (xwtm_cmp(to, next) > 0) {
240 rc = xwos_cond_wait_to(&xwscp->tx.cscond,
241 ulk, XWOS_LK_MTX, NULL,
242 next, &lkst);
243 if (-ETIMEDOUT == rc) {
244 rc = -EAGAIN;
245 }
246 } else {
247 rc = xwos_cond_wait_to(&xwscp->tx.cscond,
248 ulk, XWOS_LK_MTX, NULL,
249 to, &lkst);
250 }
251 if (__xwcc_likely(XWOK == rc)) {
252 xwos_mtx_unlock(&xwscp->tx.csmtx);
253 xwaop_write(xwu32_t, &xwscp->tx.cnt, 0, NULL);
254 xwaop_s1m(xwsq_t, &xwscp->hwifst, XWSCP_HWIFST_CONNECT, NULL, NULL);
255 xwscp_hwifal_notify(xwscp, XWSCP_HWIFNTF_CONNECT);
256 }
257 return rc;
258
259err_tx:
260 xwos_mtx_unlock(&xwscp->tx.csmtx);
261err_mtx_lock_to:
262 return rc;
263}

281xwer_t xwscp_connect(struct xwscp * xwscp, xwtm_t to)
282{
283 xwer_t rc;
284

286

305static __xwmd_code
306xwer_t xwscp_finish_tx(struct xwscp * xwscp, union xwscp_slot * slot)
307{
308 xwu8_t ack;
309 xwu8_t rmtid;
310 xwer_t rc;
311
312 ack = xwscp->tx.remote.ack;
313 rmtid = xwscp->tx.remote.id;
314
315 if (rmtid != slot->tx.frm.head.id) {
316 xwaop_c0m(xwsq_t, &xwscp->hwifst, XWSCP_HWIFST_CONNECT, NULL, NULL);
317 xwscplogf(DEBUG, "[txthd] Remote ACK ID error!\n");
318 rc = -ECONNRESET;
319 } else {
320 switch (ack) {
321 case XWSCP_ACK_OK:
322 rc = xwscp_callback_rc[ack];
323 xwaop_add(xwu32_t, &xwscp->tx.cnt, 1, NULL, NULL);
324 break;
325 case XWSCP_ACK_EALREADY:
326 rc = xwscp_callback_rc[XWSCP_ACK_OK];
327 xwaop_add(xwu32_t, &xwscp->tx.cnt, 1, NULL, NULL);
328 break;
329 case XWSCP_ACK_ECONNRESET:
330 xwaop_c0m(xwsq_t, &xwscp->hwifst, XWSCP_HWIFST_CONNECT,
331 NULL, NULL);
332 rc = -ECONNRESET;
333 break;
334 case XWSCP_ACK_NOMEM:
335 xwscplogf(WARNING, "[txthd] Remote has no memory!\n");
336 rc = -ENOMEM;
337 break;
338 default:
339 XWSCP_BUG();
340 rc = -EBUG;
341 break;
342 }
343 }
344 return rc;
345}

347static __xwmd_code
348xwer_t xwscp_tx_once(struct xwscp * xwscp,
349 const xwu8_t data[], xwsz_t * size, xwu8_t qos,
350 xwtm_t to)
351{
352 union xwos_ulock ulk;
353 union xwscp_slot * slot;
354 xwsq_t lkst;
355 xwtm_t next;
356 xwer_t rc;
357
358 rc = xwscp_fmt_msg(xwscp, data, *size, qos, &slot);
359 if (__xwcc_unlikely(rc < 0)) {
360 goto err_fmt_msg;
361 }
362 ulk.osal.mtx = &xwscp->tx.csmtx;
363 rc = xwos_mtx_lock_to(&xwscp->tx.csmtx, to);
364 if (__xwcc_unlikely(rc < 0)) {
365 goto err_mtx_lock_to;
366 }
367 xwaop_s1m(xwsq_t, &xwscp->hwifst, XWSCP_HWIFST_TX, NULL, NULL);
368 rc = xwscp_hwifal_tx(xwscp, (xwu8_t *)&slot->tx.frm, slot->tx.frmsize);
369 if (__xwcc_unlikely(rc < 0)) {
370 goto err_iftx;
371 }
372 next = xwtm_ft(XWSCP_RETRY_PERIOD);
373 if (xwtm_cmp(to, next) > 0) {
374 rc = xwos_cond_wait_to(&xwscp->tx.cscond,
375 ulk, XWOS_LK_MTX, NULL,
376 next, &lkst);
377 if (-ETIMEDOUT == rc) {
378 rc = -EAGAIN;
379 }
380 } else {
381 rc = xwos_cond_wait_to(&xwscp->tx.cscond,
382 ulk, XWOS_LK_MTX, NULL,
383 to, &lkst);
384 }
385 if (__xwcc_likely(XWOK == rc)) {
386 xwos_mtx_unlock(&xwscp->tx.csmtx);
387 rc = xwscp_finish_tx(xwscp, slot);
388 } else {
389 xwaop_c0m(xwsq_t, &xwscp->hwifst, XWSCP_HWIFST_TX, NULL, NULL);
390 }
391 return rc;
392
393err_iftx:
394 xwos_mtx_unlock(&xwscp->tx.csmtx);
395err_mtx_lock_to:
396err_fmt_msg:
397 return rc;
398}

423__xwmd_api
424xwer_t xwscp_tx(struct xwscp * xwscp,
425 const xwu8_t data[], xwsz_t * size,
426 xwu8_t qos, xwtm_t to)
427{
428 xwsq_t hwifst;
429 xwer_t rc;
430
431 XWSCP_VALIDATE((xwscp), "nullptr", -EFAULT);
432 XWSCP_VALIDATE((size), "nullptr", -EFAULT);
433 XWSCP_VALIDATE((*size <= XWSCP_SDU_MAX_SIZE), "msg2long", -E2BIG);
434
435 rc = xwos_mtx_lock_to(&xwscp->tx.mtx, to);
436 if (__xwcc_unlikely(rc < 0)) {
437 goto err_txmtx_lock_to;
438 }
439 xwaop_read(xwsq_t, &xwscp->hwifst, &hwifst);
440 if (XWSCP_HWIFST_CONNECT & hwifst) {
441 do {
442 rc = xwscp_tx_once(xwscp, data, size, qos, to);
443 } while (-EAGAIN == rc);
444 if (-ETIMEDOUT == rc) {
445 xwscp_hwifal_notify(xwscp, XWSCP_HWIFNTF_NETUNREACH);
446 rc = -ENETUNREACH;
447 }
448 } else {
449 rc = -ENOTCONN;
450 }
451 xwos_mtx_unlock(&xwscp->tx.mtx);
452 return rc;
453
454err_txmtx_lock_to:
455 return rc;
456}

478{
479 xwer_t rc;

481 xwu8_t * sdupos;
482 xwsz_t sdusize;
483 xwsz_t realsize;

496 memcpy(buf, sdupos, realsize);
497 xwmm_bma_free(xwscp->mempool, slot);
498 *size = realsize;
499 return XWOK;
500
501err_sem_wait_to:
502 return rc;
503}
