27#define XWBOP_BIT(n) ((xwbmp_t)1 << (n))
28#define XWBOP_BMP_MASK(n) ((xwbmp_t)1 << (xwbmp_t)((n) % BITS_PER_XWBMP_T))
29#define XWBOP_BMP(n) ((n) / BITS_PER_XWBMP_T)
30#define BITS_PER_XWU8_T 8
31#define XWBOP_DIV_ROUND(n, d) ((n) / (d))
32#define XWBOP_DIV_ROUND_UP(n, d) (((n) + (d) - 1) / (d))
33#define XWBOP_SHIFT_ROUND(n, s) ((n) >> (s))
34#define XWBOP_SHIFT_ROUND_UP(n, s) (((n) + (1ULL << (s)) - 1ULL) >> (s))
35#define BITS_TO_XWU8_T(n) XWBOP_DIV_ROUND_UP(n, BITS_PER_XWU8_T)
36#define BITS_TO_XWBMP_T(n) XWBOP_DIV_ROUND_UP(n, BITS_PER_XWU8_T * sizeof(xwbmp_t))
37#define XWBOP_ROUND(x, n) ((x) & (~((n) - 1U)))
38#define XWBOP_ALIGN(x, n) (((x) + ((n) - 1U)) & (~((n) - 1U)))
39#define XWBOP_TBIT(x, n) (((x) >> (n)) & 1U)
