54__xwlib_code
55xwsq_t xwos_objtik_get(void)
56{
57 xwid_t cpu;
58 atomic_xwsq_t * objtik;
59 xwsq_t curr, max, ov;
60 xwer_t rc;
61
62 cpu = xwos_skd_id_lc();
63 objtik = &xwos_objtik[cpu];
64 while (true) {
65 curr = xwaop_load(xwsq_t, objtik, xwaop_mo_relaxed);
66 max = (curr & (~XWOS_OBJTIK_CHUNK_MSK)) + XWOS_OBJTIK_CHUNK_MSK;
67 rc = xwaop_tlt_then_add(xwsq_t, objtik, max, 1, NULL, &ov);
68 if (rc < 0) {
69 xwreg_t cpuirq;
70 xwsq_t chunk;
71
72 xwos_cpuirq_save_lc(&cpuirq);
73 curr = xwaop_load(xwsq_t, objtik, xwaop_mo_relaxed);
74 if (curr == max) {
75 ov = curr;
76 xwaop_add(xwsq_t, &xwos_objtik_dispatcher,
77 XWOS_OBJTIK_CHUNK, NULL, &chunk);
78 xwaop_store(xwsq_t, objtik, xwaop_mo_relaxed, chunk);
79 xwos_cpuirq_restore_lc(cpuirq);
80 break;
81 } else {
82 xwos_cpuirq_restore_lc(cpuirq);
83 }
84 } else {
85 break;
86 }
87 }
88 return ov;
89}

124xwer_t xwos_object_activate(struct xwos_object * obj, xwobj_gc_f gcfunc)
125{
126 xwer_t rc;
127
128 rc = xwaop_teq_then_add(xwsq_t, &obj->refcnt, 0, 1, NULL, NULL);
129 if (__xwcc_likely(XWOK == rc)) {
130 obj->tik = xwos_objtik_get();
131 obj->magic = XWOS_OBJ_MAGIC;
132 obj->gcfunc = gcfunc;
133 } else {
134 rc = -EOBJACTIVE;
135 }
136 return rc;

152static __xwlib_code
153xwer_t xwos_object_acquire_refaop_tst(const void * ov, void * arg)
154{
155 const xwsq_t * o;
156 xwobj_d * objd;
157 xwer_t rc;
158
159 o = (const xwsq_t *)ov;
160 objd = arg;
161 if ((XWOS_OBJ_MAGIC != objd->obj->magic) ||
162 (objd->tik != objd->obj->tik)) {
163 rc = -EACCES;
164 } else if (*o < 1) {
165 rc = -EOBJDEAD;
166 } else {
167 rc = XWOK;
168 }
169 return rc;
