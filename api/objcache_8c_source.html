60__xwos_code
61xwer_t xwmm_mempool_objcache_init(struct xwmm_mempool_objcache * oc,
62 struct xwmm_mempool_page_allocator * pa,
63 const char * name,
64 xwsz_t objsize, xwsz_t alignment, xwsq_t pg_order,
65 ctor_f ctor, dtor_f dtor)
66{
67 xwsz_t nr, pgsize;
68
69/*
70 * ----------------------------
71 * | struct xwmm_mempool_page |
72 * -----------------------------------------------------------------------------
73 * | attr.objcache.objhead |->| obj | a | obj | a | obj | a | ...
74 * |--------------------------|-------------------------------------------------
75 * | ... | | ^ | ^ | ^
76 * |--------------------------| |_______| |_______| |_______|
77 * obj: object
78 * a: alignment area
79 */
80
81 objsize = XWBOP_ALIGN(objsize, alignment);
82 if (ctor) {
83 xwu8_t obj[objsize];
84 ctor((void *)obj);
85 oc->backup = *((xwptr_t *)obj);
86 } else {
87 oc->backup = 0;
88 }
89
90 pgsize = pa->pgsize << pg_order;
91 nr = pgsize / objsize;
92
93 oc->name = name;
94 oc->pa = pa;
95 oc->objsize = objsize;
96 oc->alignment = alignment;
97 oc->pg_order = pg_order;
98 oc->pg_objnr = nr;
99 oc->ctor = ctor;
100 oc->dtor = dtor;
101 oc->capacity = 0;
102 oc->idleness = 0;
103 oc->reserved = 0;
104 xwlib_bclst_init_head(&oc->page_list.full);
105 xwlib_bclst_init_head(&oc->page_list.available);
106 xwlib_bclst_init_head(&oc->page_list.idle);
107 xwos_sqlk_init(&oc->page_list.lock);
108 oc->i_a.malloc = xwmm_mempool_objcache_i_a_malloc;
109 oc->i_a.free = xwmm_mempool_objcache_i_a_free;
110
111 return XWOK;
112}

119static __xwos_code
120void xwmm_mempool_objcache_page_init(struct xwmm_mempool_objcache * oc,
121 struct xwmm_mempool_page * pg)
122{
123 xwsz_t i, loop;
124 xwptr_t curr, next;
125
126 xwlib_bclst_init_node(&pg->attr.objcache.node);
127 pg->attr.objcache.refcnt = 0;
128 pg->data.value = oc->objsize;
129
130 /* 构造所有对象 */
131 if (oc->ctor) {
132 curr = pg->mapping;
133 next = curr;
134 loop = oc->pg_objnr;
135 for (i = 0; i < loop; i++) {
136 next += oc->objsize;
137 oc->ctor((void *)curr);
138 curr = next;
139 }
140 }
141
142 /* 初始化无锁队列 */
143 xwlib_lfq_init(&pg->attr.objcache.objhead);
144 curr = pg->mapping;
145 next = curr;
146 loop = oc->pg_objnr;
147 for (i = 0; i < loop; i++) {
148 next += oc->objsize;
149 xwlib_lfq_init((atomic_xwlfq_t *)curr);
150 xwlib_lfq_push(&pg->attr.objcache.objhead,
151 (atomic_xwlfq_t *)curr);
152 curr = next;
153 }
154}

162static __xwos_code
163xwsz_t xwmm_mempool_objcache_free_idle_page(struct xwmm_mempool_objcache * oc,
164 xwsz_t nr)
165{
166 struct xwmm_mempool_page * pg;
167 xwreg_t flag;
168 xwsz_t real;
169
170 real = 0;
171 while (real < nr) {
172 xwos_sqlk_wr_lock_cpuirqsv(&oc->page_list.lock, &flag);
173 if (xwlib_bclst_tst_empty(&oc->page_list.idle)) {
174 xwos_sqlk_wr_unlock_cpuirqrs(&oc->page_list.lock, flag);
175 break;
176 }
177 pg = xwlib_bclst_last_entry(&oc->page_list.idle,
178 struct xwmm_mempool_page,
179 attr.objcache.node);
180 xwlib_bclst_del_init(&pg->attr.objcache.node);
181 xwos_sqlk_wr_unlock_cpuirqrs(&oc->page_list.lock, flag);
182 xwmm_mempool_page_free(oc->pa, pg);
183 xwaop_sub(xwsz_t, &oc->capacity, oc->pg_objnr, NULL, NULL);
184 xwaop_sub(xwsz_t, &oc->idleness, oc->pg_objnr, NULL, NULL);
185 real++;
186 }
187 return real;
188}

198static __xwos_code
199void xwmm_mempool_objcache_page_put(struct xwmm_mempool_objcache * oc,
200 struct xwmm_mempool_page * pg)
201{
202 xwreg_t flag;
203
204 xwos_sqlk_wr_lock_cpuirqsv(&oc->page_list.lock, &flag);
205 XWOS_BUG_ON(0 == pg->attr.objcache.refcnt);
206 pg->attr.objcache.refcnt--;
207 xwlib_bclst_del_init(&pg->attr.objcache.node);
208 if (0 == pg->attr.objcache.refcnt) {
209 xwlib_bclst_add_head(&oc->page_list.idle,
210 &pg->attr.objcache.node);
211 } else {
212 xwlib_bclst_add_head(&oc->page_list.available,
213 &pg->attr.objcache.node);
214 }
215 xwos_sqlk_wr_unlock_cpuirqrs(&oc->page_list.lock, flag);
216}

226static __xwos_code
227xwer_t xwmm_mempool_objcache_page_get(struct xwmm_mempool_objcache * oc,
228 struct xwmm_mempool_page ** pgbuf)
229{
230 struct xwmm_mempool_page * pg;
231 xwreg_t flag;
232 xwer_t rc;
233
234 xwos_sqlk_wr_lock_cpuirqsv(&oc->page_list.lock, &flag);
235 if (!xwlib_bclst_tst_empty(&oc->page_list.available)) {
236 pg = xwlib_bclst_first_entry(&oc->page_list.available,
237 struct xwmm_mempool_page,
238 attr.objcache.node);
239 xwlib_bclst_del_init(&pg->attr.objcache.node);
240 xwos_sqlk_wr_unlock_cpuirqrs(&oc->page_list.lock, flag);
241 } else if (!xwlib_bclst_tst_empty(&oc->page_list.idle)) {
242 pg = xwlib_bclst_first_entry(&oc->page_list.idle,
243 struct xwmm_mempool_page,
244 attr.objcache.node);
245 xwlib_bclst_del_init(&pg->attr.objcache.node);
246 xwos_sqlk_wr_unlock_cpuirqrs(&oc->page_list.lock, flag);
247 } else {
248 xwos_sqlk_wr_unlock_cpuirqrs(&oc->page_list.lock, flag);
249 rc = xwmm_mempool_page_allocate(oc->pa, oc->pg_order, &pg);
250 if (rc < 0) {
251 goto err_pg_alloc;
252 }
253 xwaop_add(xwsz_t, &oc->capacity, oc->pg_objnr, NULL, NULL);
254 xwaop_add(xwsz_t, &oc->idleness, oc->pg_objnr, NULL, NULL);
255 xwmm_mempool_objcache_page_init(oc, pg);
256 }
257 *pgbuf = pg;
258 return XWOK;
259
260err_pg_alloc:
261 return rc;
262}

272__xwos_code
273xwer_t xwmm_mempool_objcache_alloc(struct xwmm_mempool_objcache * oc, void ** objbuf)
274{
275 struct xwmm_mempool_page * pg;
276 xwlfq_t * obj;
277 xwreg_t flag;
278 xwer_t rc;
279
280 rc = xwmm_mempool_objcache_page_get(oc, &pg);
281 if (__xwcc_unlikely(rc < 0)) {
282 goto err_page_get;
283 }
284
285 xwaop_sub(xwsz_t, &oc->idleness, 1, NULL, NULL);
286 obj = xwlib_lfq_pop(&pg->attr.objcache.objhead);
287 XWOS_BUG_ON(NULL == obj);
288 *obj = oc->backup;
289 *objbuf = (void *)obj;
290
291 xwos_sqlk_wr_lock_cpuirqsv(&oc->page_list.lock, &flag);
292 XWOS_BUG_ON(pg->attr.objcache.refcnt >= oc->pg_objnr);
293 pg->attr.objcache.refcnt++;
294 xwlib_bclst_del_init(&pg->attr.objcache.node);
295 if (pg->attr.objcache.refcnt == oc->pg_objnr) {
296 xwlib_bclst_add_tail(&oc->page_list.full, &pg->attr.objcache.node);
297 } else {
298 xwlib_bclst_add_head(&oc->page_list.available, &pg->attr.objcache.node);
299 }
300 xwos_sqlk_wr_unlock_cpuirqrs(&oc->page_list.lock, flag);
301
302 return XWOK;
303
304err_page_get:
305 return rc;
306}

316__xwos_code
317xwer_t xwmm_mempool_objcache_free(struct xwmm_mempool_objcache * oc, void * obj)
318{
319 atomic_xwlfq_t * lfq;
320 struct xwmm_mempool_page * pg;
321 xwsz_t reserved;
322 xwsz_t idleness;
323 xwptr_t offset, origin;
324 xwer_t rc;
325
326 rc = xwmm_mempool_page_find(oc->pa, obj, &pg);
327 if (rc < 0) {
328 goto err_pg_find;
329 }
330
331 offset = ((xwptr_t)obj - pg->mapping) % oc->objsize;
332 origin = (xwptr_t)obj - offset;
333 obj = (void *)origin;
334
335 if (oc->dtor) {
336 oc->dtor(obj);
337 }
338
339 lfq = (atomic_xwlfq_t *)obj;
340 xwlib_lfq_push(&pg->attr.objcache.objhead, lfq);
341
342 xwmm_mempool_objcache_page_put(oc, pg);
343 xwaop_add(xwsz_t, &oc->idleness, 1, &idleness, NULL);
344 reserved = xwaop_load(xwsz_t, &oc->reserved, xwaop_mo_relaxed);
345
346 if (reserved + oc->pg_objnr <= idleness) {
347 xwsz_t nr;
348
349 nr = (idleness - reserved) / oc->pg_objnr;
350 xwmm_mempool_objcache_free_idle_page(oc, nr);
351 }
352
353 return XWOK;
354
355err_pg_find:
356 return rc;
357}

367__xwos_code
368xwer_t xwmm_mempool_objcache_reserve(struct xwmm_mempool_objcache * oc,
369 xwsz_t reserved)
370{
371 struct xwmm_mempool_page * pg;
372 xwsz_t capacity;
373 xwsz_t idleness;
374 xwreg_t flag;
375 xwer_t rc;
376
377 reserved = XWBOP_DIV_ROUND_UP(reserved, oc->pg_objnr);
378 xwaop_write(xwsz_t, &oc->reserved, reserved, NULL);
379 capacity = xwaop_load(xwsz_t, &oc->capacity, xwaop_mo_relaxed);
380 idleness = xwaop_load(xwsz_t, &oc->idleness, xwaop_mo_relaxed);
381 if (capacity < reserved) {
382 rc = xwmm_mempool_page_allocate(oc->pa, oc->pg_order, &pg);
383 if (rc < 0) {
384 goto err_pg_alloc;
385 }
386 xwaop_add(xwsz_t, &oc->capacity, oc->pg_objnr, NULL, NULL);
387 xwaop_add(xwsz_t, &oc->idleness, oc->pg_objnr, NULL, NULL);
388 xwmm_mempool_objcache_page_init(oc, pg);
389 xwos_sqlk_wr_lock_cpuirqsv(&oc->page_list.lock, &flag);
390 xwlib_bclst_add_tail(&oc->page_list.idle, &pg->attr.objcache.node);
391 xwos_sqlk_wr_unlock_cpuirqrs(&oc->page_list.lock, flag);
392 } else if (reserved + oc->pg_objnr < idleness) {
393 xwsz_t nr;
394
395 nr = (idleness - reserved) / oc->pg_objnr;
396 xwmm_mempool_objcache_free_idle_page(oc, nr);
397 } else {
398 }
399 return XWOK;
400
401err_pg_alloc:
402 return rc;
403}
