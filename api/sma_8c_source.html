70xwer_t xwmm_sma_alloc(struct xwmm_sma * sa, xwsz_t size, xwsz_t aligned,
71 void ** membuf)
72{
73 xwsq_t cpos, npos;
74 xwsq_t tmp;
75 xwer_t rc;
76
77 XWOS_VALIDATE((sa), "nullptr", -EFAULT);
78 XWOS_VALIDATE((membuf), "nullptr", -EFAULT);
79
80 if (__xwcc_unlikely(0 == size)) {
81 rc = -EINVAL;
82 *membuf = NULL;
83 goto err_inval;
84 }
85
86 if (aligned & XWMM_UNALIGNED_MASK) {
87 rc = -EINVAL;
88 *membuf = NULL;
89 goto err_inval;
90 }
91
92 if (0 == aligned) {
93 aligned = XWMM_ALIGNMENT;
94 }/* else {} */
95
96 size = XWBOP_ALIGN(size, XWMM_ALIGNMENT);
97 do {
98 cpos = xwaop_load(xwsq_t, &sa->pos, xwaop_mo_acquire);
99 tmp = XWBOP_ALIGN((sa->zone.origin + cpos), aligned);
100 npos = size + tmp - sa->zone.origin;
101 if (npos > sa->zone.size) {
102 rc = -ENOMEM;
103 *membuf = NULL;
104 goto err_nomem;
105 }
106 *membuf = (void *)tmp;
107 } while (xwaop_teq_then_write(xwsq_t, &sa->pos, cpos, npos, NULL));
108 return XWOK;
109
110err_nomem:
111err_inval:
112 return rc;
113}
