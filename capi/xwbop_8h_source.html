30#define XWBOP_DIV_ROUND(n, d) ((n) / (d))
31#define XWBOP_DIV_ROUND_UP(n, d) (((n) + (d) - 1U) / (d))
32#define XWBOP_SHIFT_ROUND(n, s) ((n) >> (s))
