116 if ((xwbmp_t)0 == msk) {
117 msk = (xwbmp_t)(~(xwbmp_t)0);
118 }

120 i--;

122 o &= msk;
123 if ((xwbmp_t)0 != o) {
124 pos = xwbop_fls(xwbmp_t, o);
125 break;
126 }

131 break;
132 } else {

134 n = o & (xwbmp_t)(~m);

136 }
