85__xwos_api
86xwer_t xwmm_mempool_page_allocator_init(struct xwmm_mempool_page_allocator * pa,
87 const char * name,
88 xwptr_t origin, xwsz_t size, xwsz_t pgsize,
89 struct xwmm_mempool_page_odrbtree * odrbtree,
90 struct xwmm_mempool_page * pgarray)
91{
92 xwsz_t i;
93 xwssq_t order;
94 xwsz_t nr;
95 xwer_t rc;
96
97 if (size < pgsize) {
98 rc = -E2SMALL;
99 goto err_mem2small;
100 }
101 // cppcheck-suppress [misra-c2012-10.8]
102 if ((xwsz_t)0 != (pgsize & (xwsz_t)XWMM_UNALIGNED_MASK)) {
103 rc = -EALIGN;
104 goto err_aligned;
105 }
106 nr = size / pgsize;
107 order = xwbop_fls(xwsz_t, nr);
108 if ((order < 0) || (nr & ((1U << (xwsz_t)order) - 1U))) {
109 rc = -EALIGN;
110 goto err_aligned;
111 }
112
113 pa->zone.origin = origin;
114 pa->zone.size = size;
115 pa->name = name;
116 pa->pgsize = pgsize;
117 pa->max_order = (xwsq_t)order;
118 pa->odrbtree = odrbtree;
119 pa->pgarray = pgarray;
120
121 for (i = 0; i <= pa->max_order; i++) {
122 xwlib_rbtree_init(&pa->odrbtree[i].tree);
123 pa->odrbtree[i].leftmost = NULL;
124 pa->odrbtree[i].owner = pa;
125 pa->odrbtree[i].order = i;
126 xwos_sqlk_init(&pa->odrbtree[i].lock);
127 }
128 for (i = 0; i < nr; i++) {
129 pa->pgarray[i].order = XWMM_MEMPOOL_PAGE_ORDER_CMB;
130 pa->pgarray[i].mapping = XWMM_MEMPOOL_PAGE_MAPPING_INVAILD;
131 pa->pgarray[i].attr.free.seq = i;
132 xwlib_rbtree_init_node(&pa->pgarray[i].attr.free.rbnode);
133 pa->pgarray[i].data.value = 0;
134 }
135 // cppcheck-suppress [misra-c2012-17.7]
136 xwmm_mempool_page_odrbtree_add(&pa->odrbtree[pa->max_order],
137 &pa->pgarray[0]);
138
139 pa->i_a.malloc = xwmm_mempool_page_i_a_malloc;
140 pa->i_a.free = xwmm_mempool_page_i_a_free;
141
142 return XWOK;
143
144err_aligned:
145err_mem2small:
146 return rc;
147}

158__xwos_api
159xwer_t xwmm_mempool_page_allocate(struct xwmm_mempool_page_allocator * pa,
160 xwsq_t order,
161 struct xwmm_mempool_page ** pgbuf)
162{
163 xwsq_t odr;
164 struct xwmm_mempool_page_odrbtree * ot;
165 struct xwmm_mempool_page * pg;
166 xwptr_t origin;
167 xwer_t rc;
168
169 rc = -ENOMEM;
170 for (odr = order; odr <= pa->max_order; odr++) {
171 ot = &pa->odrbtree[odr];
172 pg = xwmm_mempool_page_odrbtree_choose(ot);
173 if (NULL != pg) {
174 xwmm_mempool_page_divide_page(pa, pg, order, ot);
175 origin = pa->zone.origin + (pg->attr.free.seq * pa->pgsize);
176 pg->mapping = origin;
177 *pgbuf = pg;
178 rc = XWOK;
179 break;
180 }
181 }
182 return rc;
183}

221static __xwos_code
222void xwmm_mempool_page_divide_page(struct xwmm_mempool_page_allocator * pa,
223 struct xwmm_mempool_page * pg,
224 xwsq_t target_odr,
225 struct xwmm_mempool_page_odrbtree * ot)
226{
227 xwsq_t pg_odr;
228 xwsq_t pg_seq;
229 xwsq_t pg_offset;
230 struct xwmm_mempool_page * buddy;
231
232 pg_odr = pg->order;
233 pg_seq = pg->attr.free.seq;
234 pg_offset = 1U << pg_odr;
235 while (pg_odr > target_odr) {
236 ot--;
237 pg_odr--;
238 pg_offset >>= 1;
239 buddy = &pa->pgarray[pg_seq + pg_offset];
240 // cppcheck-suppress [misra-c2012-17.7]
241 xwmm_mempool_page_odrbtree_add(ot, buddy);
242 pg->order = pg_odr;
243 }
244}

251static __xwos_code
252void xwmm_mempool_page_combine(struct xwmm_mempool_page_allocator * pa,
253 struct xwmm_mempool_page * pg)
254{
255 struct xwmm_mempool_page * buddy;
256 xwsq_t odr;
257 xwer_t rc;
258
259 odr = pg->order + 1U;
260 while (odr < pa->max_order) {
261 buddy = xwmm_mempool_page_get_buddy(pa, pg);
262 /* `buddy->order` 不能在临界区外使用，
263 因为 `buddy` 在阶红黑树上时，是共享数据。
264 但可以使用 `pg->order` ，因为 `pg` 还未加入到阶红黑树，是本地数据。*/
265 rc = xwmm_mempool_page_odrbtree_remove(&pa->odrbtree[pg->order], buddy);
266 if (rc < 0) {
267 break;
268 }
269 if (buddy > pg) {
270 buddy->order = XWMM_MEMPOOL_PAGE_ORDER_CMB;
271 pg->order = odr;
272 } else {
273 pg->order = XWMM_MEMPOOL_PAGE_ORDER_CMB;
274 buddy->order = odr;
275 pg = buddy;
276 }
277 odr++;
278 }
279 odr = pg->order;
280 // cppcheck-suppress [misra-c2012-17.7]
281 xwmm_mempool_page_odrbtree_add(&pa->odrbtree[odr], pg);
282}

292static __xwos_code
293xwer_t xwmm_mempool_page_odrbtree_add(struct xwmm_mempool_page_odrbtree * ot,
294 struct xwmm_mempool_page * pg)
295{
296 struct xwlib_rbtree * tree;
297 struct xwlib_rbtree_node ** new;
298 xwptr_t lpc;
299 struct xwlib_rbtree_node * rbn;
300 xwsq_t pgseq;
301 struct xwmm_mempool_page * b;
302 struct xwmm_mempool_page * leftmost;
303 xwreg_t flag;
304 xwer_t rc;
305
306 tree = &ot->tree;
307 pgseq = pg->attr.free.seq;
308 /* 即将加入的页不在阶红黑树中，是本地数据，对 `pg->mapping` 的读可在锁外。*/
309 if (XWMM_MEMPOOL_PAGE_MAPPING_FREE == pg->mapping) {
310 rc = -EEXIST;
311 goto err_exist;
312 }
313
314 pg->order = ot->order;
315 xwos_sqlk_wr_lock_cpuirqsv(&ot->lock, &flag);
316 /* 当页被加入到阶红黑树中，就变成共享数据，对 `pg->mapping` 的写只在锁内。*/
317 pg->mapping = XWMM_MEMPOOL_PAGE_MAPPING_FREE;
318 new = &tree->root;
319 lpc = (xwptr_t)new;
320 leftmost = ot->leftmost;
321 if (NULL == leftmost) { /* rbtree is empty. */
322 ot->leftmost = pg;
323 } else if (pgseq < leftmost->attr.free.seq) {
324 new = &(leftmost->attr.free.rbnode.left);
325 lpc = (xwptr_t)new;
326 ot->leftmost = pg;
327 } else {
328 rbn = *new;
329 while (NULL != rbn) {
330 b = xwlib_rbtree_entry(rbn, struct xwmm_mempool_page,
331 attr.free.rbnode);
332 if (pgseq < b->attr.free.seq) {
333 new = &rbn->left;
334 lpc = (xwptr_t)new;
335 rbn = rbn->left;
336 } else if (pgseq > b->attr.free.seq) {
337 new = &rbn->right;
338 lpc = (xwptr_t)new | (xwptr_t)XWLIB_RBTREE_POS_RIGHT;
339 rbn = rbn->right;
340 } else {
341 XWOS_BUG();
342 }
343 }
344 }
345 xwlib_rbtree_link(&pg->attr.free.rbnode, lpc);
346 xwlib_rbtree_insert_color(tree, &pg->attr.free.rbnode);
347 xwos_sqlk_wr_unlock_cpuirqrs(&ot->lock, flag);
348 return XWOK;
349
350err_exist:
351 return rc;
352}

363static __xwos_code
364xwer_t xwmm_mempool_page_odrbtree_remove(struct xwmm_mempool_page_odrbtree * ot,
365 struct xwmm_mempool_page * pg)
366{
367 struct xwlib_rbtree * tree;
368 struct xwlib_rbtree_node * s;
369 xwreg_t flag;
370 xwer_t rc;
371
372 tree = &ot->tree;
373 if (pg->order != ot->order) {
374 rc = -ESRCH;
375 goto err_odrerr;
376 }
377
378 xwos_sqlk_wr_lock_cpuirqsv(&ot->lock, &flag);
379 /* 即将被删除的页在阶红黑树中，是共享数据。对 `pg->mapping` 的读只可在锁内。*/
380 if (XWMM_MEMPOOL_PAGE_MAPPING_FREE != pg->mapping) {
381 rc = -EBUSY;
382 goto err_notfree;
383 }
384 pg->mapping = XWMM_MEMPOOL_PAGE_MAPPING_INVAILD;
385 if (pg == ot->leftmost) {
386 s = pg->attr.free.rbnode.right;
387 if (NULL == s) {
388 s = xwlib_rbtree_get_parent(&pg->attr.free.rbnode);
389 }
390 if (s != (struct xwlib_rbtree_node *)&tree->root) {
391 ot->leftmost = xwlib_rbtree_entry(s,
392 struct xwmm_mempool_page,
393 attr.free.rbnode);
394 } else {
395 ot->leftmost = NULL;
396 }
397 }
398 xwlib_rbtree_remove(tree, &pg->attr.free.rbnode);
399 xwos_sqlk_wr_unlock_cpuirqrs(&ot->lock, flag);
400 xwlib_rbtree_init_node(&pg->attr.free.rbnode);
401 return XWOK;
402
403err_notfree:
404 xwos_sqlk_wr_unlock_cpuirqrs(&ot->lock, flag);
405err_odrerr:
406 return rc;
407}

414static __xwos_code
415struct xwmm_mempool_page *
416xwmm_mempool_page_odrbtree_choose(struct xwmm_mempool_page_odrbtree * ot)
417{
418 struct xwmm_mempool_page * leftmost;
419 xwreg_t flag;
420 struct xwlib_rbtree * tree;
421 struct xwlib_rbtree_node * s;
422
423 tree = &ot->tree;
424 xwos_sqlk_wr_lock_cpuirqsv(&ot->lock, &flag);
425 leftmost = ot->leftmost;
426 if (NULL == leftmost) {
427 xwos_sqlk_wr_unlock_cpuirqrs(&ot->lock, flag);
428 goto err_empty;
429 }
430 /* 即将被选择的页在阶红黑树中，是共享数据。对 `pg->mapping` 的写只可在锁内。*/
431 leftmost->mapping = XWMM_MEMPOOL_PAGE_MAPPING_INVAILD;
432 s = leftmost->attr.free.rbnode.right;
433 if (NULL == s) {
434 s = xwlib_rbtree_get_parent(&leftmost->attr.free.rbnode);
435 }
436 if (s != (struct xwlib_rbtree_node *)&tree->root) {
437 ot->leftmost = xwlib_rbtree_entry(s, struct xwmm_mempool_page,
438 attr.free.rbnode);
439 } else {
440 ot->leftmost = NULL;
441 }
442 xwlib_rbtree_remove(tree, &leftmost->attr.free.rbnode);
443 xwos_sqlk_wr_unlock_cpuirqrs(&ot->lock, flag);
444
445 xwlib_rbtree_init_node(&leftmost->attr.free.rbnode);
446 return leftmost;
447
448err_empty:
449 return NULL;
450}

521{

524

534 rc = XWOK;

548static __xwos_code
549xwer_t xwmm_mempool_page_i_a_malloc(void * this, xwsz_t size, void ** membuf)
550{
551 struct xwmm_mempool_page_allocator * pa;
552 struct xwmm_mempool_page * pg;
553 xwssq_t order;
554 xwsz_t nr;
555 xwer_t rc;
556
557 pa = this;
558 nr = size / pa->pgsize;
559 order = xwbop_fls(xwsz_t, nr);
560 if ((order < 0) || ((pa->pgsize << (xwsz_t)order) < size)) {
561 order++;
562 }
563 rc = xwmm_mempool_page_allocate(pa, (xwsq_t)order, &pg);
564 if (XWOK == rc) {
565 pg->data.value = size;
566 *membuf = (void *)pg->mapping;
567 }
568 return rc;
569}
