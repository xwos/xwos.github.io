170xwer_t xwds_cantrcv_xxx10xx_drv_set_opmode(struct xwds_cantrcv * cantrcv,
171 xwsq_t opmode)
172{
173 struct xwds_cantrcv_xxx10xx * xxx10xx;
174 const struct xwds_resource_gpio * gpiorsc;
175 xwer_t rc;
176
177 xxx10xx = xwds_cast(struct xwds_cantrcv_xxx10xx *, cantrcv);
178 gpiorsc = xxx10xx->gpiorsc.stb;
179 switch (opmode) {
180 case XWDS_CANTRCV_OPMODE_NORMAL:
181 rc = xwds_gpio_reset(gpiorsc->soc,
182 gpiorsc->port,
183 gpiorsc->pinmask);
184 if (XWOK == rc) {
185 xxx10xx->cantrcv.wkuprs = XWDS_CANTRCV_WKUPRS_BY_PIN;
186 }
187 break;
188 case XWDS_CANTRCV_OPMODE_SLEEP:
189 case XWDS_CANTRCV_OPMODE_STANDBY:
190 rc = xwds_gpio_set(gpiorsc->soc,
191 gpiorsc->port,
192 gpiorsc->pinmask);
193 break;
194 default:
195 rc = -ERANGE;
196 break;
197 }
198 return rc;
199}
