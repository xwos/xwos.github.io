128{
129 xwer_t rc;
130
131 if (spip->bus) {
132 rc = xwds_spim_grab(spip->bus);
133 } else {
134 rc = -EHOSTUNREACH;
135 }
136 if (rc < 0) {
137 goto err_spim_grab;
138 }
139 rc = xwds_device_vop_start(&spip->dev);
140 if (rc < 0) {
141 goto err_dev_vop_start;
142 }
143 return XWOK;
144
145err_dev_vop_start:
146 xwds_spim_put(spip->bus);
147err_spim_grab:
148 return rc;
149}
