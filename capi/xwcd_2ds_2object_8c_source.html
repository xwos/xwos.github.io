171__xwds_code
172xwer_t xwds_obj_add(struct xwds * ds, struct xwds_object * obj)
173{
174 xwreg_t cpuirq;
175 xwer_t rc;
176
177 xwos_sqlk_wr_lock_cpuirqsv(&ds->devlistlock, &cpuirq);
178 if (obj->ds) {
179 rc = -EPERM;
180 } else {
181 xwlib_bclst_add_tail(&ds->devhead, &obj->node); /* LIFO */
182 obj->ds = ds;
183 rc = XWOK;
184 }
185 xwos_sqlk_wr_unlock_cpuirqrs(&ds->devlistlock, cpuirq);
186 return rc;
187}

189__xwds_code
190xwer_t xwds_obj_del(struct xwds * ds, struct xwds_object * obj)
191{
192 xwreg_t cpuirq;
193 xwer_t rc;
194
195 xwos_sqlk_wr_lock_cpuirqsv(&ds->devlistlock, &cpuirq);
196 if (obj->ds != ds) {
197 rc = -EOWNER;
198 } else {
199 xwlib_bclst_del_init(&obj->node); /* LIFO */
200 obj->ds = NULL;
201 rc = XWOK;
202 }
203 xwos_sqlk_wr_unlock_cpuirqrs(&ds->devlistlock, cpuirq);
204 return rc;
205}
